"""
commutation.py
==============
Sterbetafel-Zugriff, Kommutationswerte, mathematische Hilfsfunktionen.

DARF NICHT actuarial.py importieren.
"""
from __future__ import annotations

import os
import xml.etree.ElementTree as ET
from functools import lru_cache
from typing import Dict, Tuple

# ---------------------------------------------------------------------------
# Konstanten (aus VBA mConstants)
# ---------------------------------------------------------------------------
RUND_LX: int = 16
RUND_TX: int = 16
RUND_DX: int = 16
RUND_CX: int = 16
RUND_NX: int = 16
RUND_MX: int = 16
RUND_RX: int = 16
MAX_ALTER: int = 123

# ---------------------------------------------------------------------------
# Sterbetafel laden
# ---------------------------------------------------------------------------
_TAFELN_XML = os.path.join(os.path.dirname(__file__), "tafeln.xml")

_TAFEL_CACHE: Dict[str, Tuple[float, ...]] = {}


def _load_tafel(tafel_id: str) -> Tuple[float, ...]:
    """Lädt qx-Vektor für tafel_id aus tafeln.xml (0..MAX_ALTER)."""
    if tafel_id in _TAFEL_CACHE:
        return _TAFEL_CACHE[tafel_id]
    tree = ET.parse(_TAFELN_XML)
    root = tree.getroot()
    for tafel_el in root.findall("Tafel"):
        if tafel_el.get("id") == tafel_id:
            qx_map: Dict[int, float] = {}
            for qx_el in tafel_el.findall("qx"):
                alter = int(qx_el.get("alter"))
                qx_map[alter] = float(qx_el.text)
            vec = tuple(qx_map[i] for i in range(MAX_ALTER + 1))
            _TAFEL_CACHE[tafel_id] = vec
            return vec
    raise NotImplementedError(
        f"Sterbetafel '{tafel_id}' nicht in tafeln.xml implementiert. "
        "Bitte qx-Werte eintragen."
    )


def _tafel_id(tafel: str, sex: str) -> str:
    """Erzeugt den Tafel-Bezeichner (z.B. 'DAV1994_T_M')."""
    sex_norm = "M" if sex.upper() == "M" else "F"
    return f"{tafel.upper()}_{sex_norm}"


# ---------------------------------------------------------------------------
# lx-Vektor  (VBA: v_lx)
# ---------------------------------------------------------------------------

def _v_lx(tafel: str, sex: str) -> Tuple[float, ...]:
    qx = _load_tafel(_tafel_id(tafel, sex))
    lx = [0.0] * (MAX_ALTER + 1)
    lx[0] = 1_000_000.0
    for i in range(1, MAX_ALTER + 1):
        raw = lx[i - 1] * (1.0 - qx[i - 1])
        lx[i] = round(raw, RUND_LX)
    return tuple(lx)


# ---------------------------------------------------------------------------
# tx-Vektor  (VBA: v_tx)
# ---------------------------------------------------------------------------

def _v_tx(tafel: str, sex: str) -> Tuple[float, ...]:
    lx = _v_lx(tafel, sex)
    tx = [0.0] * (MAX_ALTER + 1)
    for i in range(MAX_ALTER):
        tx[i] = round(lx[i] - lx[i + 1], RUND_TX)
    return tuple(tx)


# ---------------------------------------------------------------------------
# Dx-Vektor  (VBA: v_Dx)
# ---------------------------------------------------------------------------

def _v_Dx(tafel: str, sex: str, zins: float) -> Tuple[float, ...]:
    lx = _v_lx(tafel, sex)
    v = 1.0 / (1.0 + zins)
    dx = [0.0] * (MAX_ALTER + 1)
    for i in range(MAX_ALTER + 1):
        dx[i] = round(lx[i] * (v ** i), RUND_DX)
    return tuple(dx)


# ---------------------------------------------------------------------------
# Cx-Vektor  (VBA: v_Cx)
# ---------------------------------------------------------------------------

def _v_Cx(tafel: str, sex: str, zins: float) -> Tuple[float, ...]:
    tx = _v_tx(tafel, sex)
    v = 1.0 / (1.0 + zins)
    cx = [0.0] * (MAX_ALTER + 1)
    for i in range(MAX_ALTER):   # MAX_ALTER bleibt 0
        cx[i] = round(tx[i] * (v ** (i + 1)), RUND_CX)
    return tuple(cx)


# ---------------------------------------------------------------------------
# Nx-Vektor  (VBA: v_Nx)  — rückwärts summiert
# ---------------------------------------------------------------------------

def _v_Nx(tafel: str, sex: str, zins: float) -> Tuple[float, ...]:
    dx = _v_Dx(tafel, sex, zins)
    nx = [0.0] * (MAX_ALTER + 1)
    nx[MAX_ALTER] = dx[MAX_ALTER]
    for i in range(MAX_ALTER - 1, -1, -1):
        nx[i] = round(nx[i + 1] + dx[i], RUND_NX)
    return tuple(nx)


# ---------------------------------------------------------------------------
# Mx-Vektor  (VBA: v_Mx)  — rückwärts summiert
# ---------------------------------------------------------------------------

def _v_Mx(tafel: str, sex: str, zins: float) -> Tuple[float, ...]:
    cx = _v_Cx(tafel, sex, zins)
    mx = [0.0] * (MAX_ALTER + 1)
    mx[MAX_ALTER] = cx[MAX_ALTER]   # = 0.0
    for i in range(MAX_ALTER - 1, -1, -1):
        mx[i] = round(mx[i + 1] + cx[i], RUND_MX)
    return tuple(mx)


# ---------------------------------------------------------------------------
# Öffentliche Skalar-Funktionen
# ---------------------------------------------------------------------------

@lru_cache(maxsize=None)
def Act_Dx(alter: int, sex: str, tafel: str, zins: float) -> float:
    """D_x Kommutationswert."""
    return _v_Dx(tafel, sex, zins)[alter]


@lru_cache(maxsize=None)
def Act_Nx(alter: int, sex: str, tafel: str, zins: float) -> float:
    """N_x Kommutationswert."""
    return _v_Nx(tafel, sex, zins)[alter]


@lru_cache(maxsize=None)
def Act_Mx(alter: int, sex: str, tafel: str, zins: float) -> float:
    """M_x Kommutationswert."""
    return _v_Mx(tafel, sex, zins)[alter]


@lru_cache(maxsize=None)
def Act_Cx(alter: int, sex: str, tafel: str, zins: float) -> float:
    """C_x Kommutationswert."""
    return _v_Cx(tafel, sex, zins)[alter]


# ---------------------------------------------------------------------------
# Abzugsglied  (VBA: Act_Abzugsglied)
# ---------------------------------------------------------------------------

@lru_cache(maxsize=None)
def Act_Abzugsglied(k: int, zins: float) -> float:
    """
    Abzugsglied für unterjährige Rentenzahlweise k.
    VBA-Quelle: mBarwerte / Act_Abzugsglied
    """
    if k <= 0:
        return 0.0
    total = 0.0
    for l in range(k):
        frac = l / k
        total += frac / (1.0 + frac * zins)
    return total * (1.0 + zins) / k


# ---------------------------------------------------------------------------
# Barwert-Funktionen (VBA: mBarwerte)
# ---------------------------------------------------------------------------

def Act_axn_k(alter: int, n: int, sex: str, tafel: str, zins: float, k: int) -> float:
    """
    Leibrenten-Barwert (n-jährig, k-tel-jährlich vorschüssig).
    VBA: Act_axn_k
    """
    if k <= 0:
        return 0.0
    Dx_x = Act_Dx(alter, sex, tafel, zins)
    Dx_xn = Act_Dx(alter + n, sex, tafel, zins)
    Nx_x = Act_Nx(alter, sex, tafel, zins)
    Nx_xn = Act_Nx(alter + n, sex, tafel, zins)
    abz = Act_Abzugsglied(k, zins)
    ratio = Dx_xn / Dx_x
    return (Nx_x - Nx_xn) / Dx_x - abz * (1.0 - ratio)


def Act_ax_k(alter: int, sex: str, tafel: str, zins: float, k: int) -> float:
    """
    Leibrenten-Barwert (unendlich, k-tel-jährlich vorschüssig).
    VBA: Act_ax_k
    """
    if k <= 0:
        return 0.0
    return Act_Nx(alter, sex, tafel, zins) / Act_Dx(alter, sex, tafel, zins) - Act_Abzugsglied(k, zins)


def act_nGrAx(alter: int, n: int, sex: str, tafel: str, zins: float) -> float:
    """
    Temporäre Todesfallversicherung (reine Risikoversicherung Barwert).
    VBA: Act_nGrAx
    """
    Mx_x = Act_Mx(alter, sex, tafel, zins)
    Mx_xn = Act_Mx(alter + n, sex, tafel, zins)
    Dx_x = Act_Dx(alter, sex, tafel, zins)
    return (Mx_x - Mx_xn) / Dx_x


def Act_nGrEx(alter: int, n: int, sex: str, tafel: str, zins: float) -> float:
    """
    Reiner Erlebensfallbarwert.
    VBA: Act_nGrEx
    """
    return Act_Dx(alter + n, sex, tafel, zins) / Act_Dx(alter, sex, tafel, zins)

"""
actuarial.py
============
Barwerte, Tarif- und Produktlogik, aktuarielle Zielgrößen.

Importiert commutation.py (erlaubte Richtung).
Darf NICHT von commutation.py importiert werden.
"""
from __future__ import annotations

from typing import List, Dict, Any

from commutation import (
    act_nGrAx,
    Act_Dx,
    Act_axn_k,
)
from inputs import VertragInputs
from params import TarifParameter, ratzu


# ---------------------------------------------------------------------------
# Skalare Tarifgrößen
# ---------------------------------------------------------------------------

def calc_Bxt(inp: VertragInputs, par: TarifParameter) -> float:
    """
    Netto-Beitragssatz (pro 1 VS) – jährlicher Äquivalenzbeitrag Bxt.
    Excel K5 / VBA-Name B_xt.

    Formel:
        Bxt = (nGrAx(x,n) + nEx(x,n) + gamma1*axn(x,t) + gamma2*(axn(x,n)-axn(x,t)))
              / ((1-beta1)*axn(x,t) - alpha*t)
    """
    x, n, t = inp.x, inp.n, inp.t
    sex, tafel, zins = inp.sex, par.Tafel, par.Zins
    alpha, beta1 = par.alpha, par.beta1
    gamma1, gamma2 = par.gamma1, par.gamma2

    nGrAx_val = act_nGrAx(x, n, sex, tafel, zins)
    nEx_val = Act_Dx(x + n, sex, tafel, zins) / Act_Dx(x, sex, tafel, zins)
    axn_n = Act_axn_k(x, n, sex, tafel, zins, 1)
    axn_t = Act_axn_k(x, t, sex, tafel, zins, 1)

    zaehler = nGrAx_val + nEx_val + gamma1 * axn_t + gamma2 * (axn_n - axn_t)
    nenner = (1.0 - beta1) * axn_t - alpha * t
    return zaehler / nenner


def calc_BJB(inp: VertragInputs, par: TarifParameter) -> float:
    """
    Jahresbeitrag (brutto) in EUR.
    Excel K6 = VS * Bxt
    """
    return inp.VS * calc_Bxt(inp, par)


def calc_BZB(inp: VertragInputs, par: TarifParameter) -> float:
    """
    Zahlbeitrag (monatlich/viertelj./halbj.) inkl. Ratenzuschlag.
    Excel K7 = (1+ratzu)/zw * (BJB + k)
    k hier = par.k (Policengebühr in EUR p.a.)
    """
    rz = ratzu(inp.zw)
    BJB = calc_BJB(inp, par)
    return (1.0 + rz) / inp.zw * (BJB + par.k)


def calc_Pxt(inp: VertragInputs, par: TarifParameter) -> float:
    """
    Netto-Beitragssatz (alternativer Ausdruck mit B_xt bereits bekannt).
    Excel K9.

    Formel:
        Pxt = (nGrAx(x,n) + nEx(x,n) + t*alpha*Bxt) / axn(x,t)
    """
    x, n, t = inp.x, inp.n, inp.t
    sex, tafel, zins = inp.sex, par.Tafel, par.Zins

    nGrAx_val = act_nGrAx(x, n, sex, tafel, zins)
    nEx_val = Act_Dx(x + n, sex, tafel, zins) / Act_Dx(x, sex, tafel, zins)
    axn_t = Act_axn_k(x, t, sex, tafel, zins, 1)
    B_xt = calc_Bxt(inp, par)

    return (nGrAx_val + nEx_val + t * par.alpha * B_xt) / axn_t


def calc_ratzu(inp: VertragInputs) -> float:
    """Ratenzuschlag (Excel E12)."""
    return ratzu(inp.zw)


# ---------------------------------------------------------------------------
# Verlaufstabelle (Zeilen k=0..n+flex)
# ---------------------------------------------------------------------------

def calc_verlauf(inp: VertragInputs, par: TarifParameter) -> List[Dict[str, Any]]:
    """
    Berechnet die Verlaufswerte für k = 0..50 (analog Excel-Zeilen 16..66).

    Spalten:
        k, Axn, axn, axt, kVx_bpfl, kDRx_bpfl, kVx_bfr,
        kVx_MRV, flex_Phase, StoAb, RKW, VS_bfr
    """
    x, n, t = inp.x, inp.n, inp.t
    sex, tafel, zins = inp.sex, par.Tafel, par.Zins
    VS = inp.VS
    alpha, gamma2, gamma3 = par.alpha, par.gamma2, par.gamma3
    MinAlterFlex, MinRLZFlex = par.MinAlterFlex, par.MinRLZFlex

    B_xt = calc_Bxt(inp, par)
    BJB = inp.VS * B_xt
    P_xt = calc_Pxt(inp, par)

    # axn(x,n) und axn(x,t) für Konstante in kVx_bpfl
    axn_n_0 = Act_axn_k(x, n, sex, tafel, zins, 1)
    axn_t_0 = Act_axn_k(x, t, sex, tafel, zins, 1)
    ratio_axn = axn_n_0 / axn_t_0 if axn_t_0 != 0.0 else 0.0

    # axn(x, 5) für MRV-Nenner
    axn_5_0 = Act_axn_k(x, 5, sex, tafel, zins, 1)

    rows: List[Dict[str, Any]] = []
    for step in range(51):  # k = 0 .. 50
        k_val = step

        # Axn (Spalte B)
        if k_val <= n:
            axn_rem = max(0, n - k_val)
            Axn = (
                act_nGrAx(x + k_val, axn_rem, sex, tafel, zins)
                + Act_Dx(x + n, sex, tafel, zins) / Act_Dx(x + k_val, sex, tafel, zins)
            )
        else:
            Axn = 0.0

        # axn (Spalte C)
        axn_c = Act_axn_k(x + k_val, max(0, n - k_val), sex, tafel, zins, 1)

        # axt (Spalte D)
        axt_d = Act_axn_k(x + k_val, max(0, t - k_val), sex, tafel, zins, 1)

        # kVx_bpfl (Spalte E)
        kVx_bpfl = Axn - P_xt * axt_d + gamma2 * (axn_c - ratio_axn * axt_d)

        # kDRx_bpfl (Spalte F)
        kDRx_bpfl = VS * kVx_bpfl

        # kVx_bfr (Spalte G)
        kVx_bfr = Axn + gamma3 * axn_c

        # kVx_MRV (Spalte H)
        axn_mrv = Act_axn_k(x + k_val, max(5 - k_val, 0), sex, tafel, zins, 1)
        if axn_5_0 != 0.0:
            kVx_MRV = kDRx_bpfl + alpha * t * BJB * axn_mrv / axn_5_0
        else:
            kVx_MRV = kDRx_bpfl

        # flex. Phase (Spalte I)
        flex_phase = 1 if (x + k_val >= MinAlterFlex and k_val >= n - MinRLZFlex) else 0

        # StoAb (Spalte J)
        if k_val > n or flex_phase:
            StoAb = 0.0
        else:
            StoAb = min(150.0, max(50.0, 0.01 * (VS - kDRx_bpfl)))

        # RKW (Spalte K)
        RKW = max(0.0, kVx_MRV - StoAb)

        # VS_bfr (Spalte L)
        try:
            if k_val > n:
                VS_bfr = 0.0
            elif k_val < t:
                VS_bfr = kVx_MRV / kVx_bfr if kVx_bfr != 0.0 else 0.0
            else:
                VS_bfr = float(VS)
        except (ZeroDivisionError, ValueError):
            VS_bfr = 0.0

        rows.append({
            "k": k_val,
            "Axn": Axn,
            "axn": axn_c,
            "axt": axt_d,
            "kVx_bpfl": kVx_bpfl,
            "kDRx_bpfl": kDRx_bpfl,
            "kVx_bfr": kVx_bfr,
            "kVx_MRV": kVx_MRV,
            "flex_Phase": float(flex_phase),
            "StoAb": StoAb,
            "RKW": RKW,
            "VS_bfr": VS_bfr,
        })

    return rows

    inp = DEFAULT_INP
    par = DEFAULT_PAR

    print("=" * 60)
    print("KLV-Tarifrechner – deterministischer Testlauf")
    print("=" * 60)
    print(f"  Eintrittsalter x = {inp.x}")
    print(f"  Geschlecht       = {inp.sex}")
    print(f"  Laufzeit n       = {inp.n}")
    print(f"  Beitragszeit t   = {inp.t}")
    print(f"  VS               = {inp.VS:,.2f}")
    print(f"  Zahlweise zw     = {inp.zw}")
    print(f"  Tafel            = {par.Tafel}")
    print(f"  Zins             = {par.Zins:.4%}")
    print()

    Bxt  = calc_Bxt(inp, par)
    BJB  = calc_BJB(inp, par)
    BZB  = calc_BZB(inp, par)
    Pxt  = calc_Pxt(inp, par)
    rz   = calc_ratzu(inp)

    print(f"  ratzu            = {rz:.4%}")
    print(f"  Bxt  (B_xt)      = {Bxt:.10f}")
    print(f"  BJB              = {BJB:.6f}")
    print(f"  BZB              = {BZB:.6f}")
    print(f"  Pxt  (P_xt)      = {Pxt:.10f}")
    print()

    verlauf = calc_verlauf(inp, par)
    print(f"{'k':>3}  {'Axn':>12}  {'axn':>12}  {'axt':>12}  "
          f"{'kVx_bpfl':>12}  {'kDRx_bpfl':>14}  {'RKW':>12}  {'VS_bfr':>12}")
    print("-" * 110)
    for row in verlauf:
        print(
            f"{row['k']:>3}  {row['Axn']:>12.6f}  {row['axn']:>12.6f}  "
            f"{row['axt']:>12.6f}  {row['kVx_bpfl']:>12.6f}  "
            f"{row['kDRx_bpfl']:>14.2f}  {row['RKW']:>12.2f}  "
            f"{row['VS_bfr']:>12.2f}"
        )

    print()
    print("Golden-Master-Struktur:")
    gm = golden_master_outputs()
    print(f"  Skalare:  {list(gm['scalars']['Kalkulation'].keys())}")
    print(f"  Tabellen: {len(gm['tables']['Kalkulation'])} Zeilen, "
          f"Spalten: {list(gm['tables']['Kalkulation'][0].keys())}")
    print()
    print("TEST BESTANDEN – deterministisch, keine externen Abhängigkeiten.")


if __name__ == "__main__":
    run_test()
